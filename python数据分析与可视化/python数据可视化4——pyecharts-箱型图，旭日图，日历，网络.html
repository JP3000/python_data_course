#这个代码的前提是index本身是要已经排序好了
data=[]
for i in store:
    z1={"name":i,"itemStyle": {"color": random.choice(color)},"children":[]}
    i0=[0,0]
    z2={}
    for j in p2[i].index:
        if j[0]!=i0[0]:
            i0[0]=j[0]
            if len(z2)>0:z1["children"].append(z2)
            z2={"name":j[0],"itemStyle": {"color": random.choice(color)},"children":[]}
            z3={"name":j[1],"itemStyle": {"color": random.choice(color)},"value":int(p2[i][j[0]][j[1]])}
            z2["children"].append(z3)
        else:
            z3={"name":j[1],"itemStyle": {"color": random.choice(color)},"value":int(p2[i][j[0]][j[1]])}
            z2["children"].append(z3)
    z1["children"].append(z2)
    data.append(z1)

#https://gallery.pyecharts.org/#/Calendar/calendar_base
import datetime
import random
from pyecharts import options as opts
from pyecharts.charts import Calendar


begin = datetime.date(2017, 1, 1)
end = datetime.date(2017, 12, 31)

data = [
    [str(begin + datetime.timedelta(days=i)), random.randint(1000, 25000)]
    for i in range((end - begin).days + 1)
]

C1=Calendar()
C1.add("", data, calendar_opts=opts.CalendarOpts(range_="2017"))
C1.set_global_opts(
        title_opts=opts.TitleOpts(title="Calendar-2017年微信步数情况"),
        visualmap_opts=opts.VisualMapOpts(
            max_=20000,
            min_=500,
            orient="horizontal",
            is_piecewise=True,
            pos_top="230px",
            pos_left="100px",
        ),
    )
C1.render_notebook()
